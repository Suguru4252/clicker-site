// Состояние игры let gameState = { player: { name: "Выживший", level: 1, health: 100, maxHealth: 100, hunger: 100, thirst: 100, exp: 0, expToNext: 100 }, inventory: { "Древесина": 15, "Камень": 10, "Ягоды": 8, "Ткань": 3, "Каменный нож": 1, "Еда": 2, "Лук": 0, "Топор": 0 } }; // Животные для охоты const animals = [ { name: "🐗 Кабан", health: 30, damage: 10, drops: { "Еда": 2, "Ткань": 1 } }, { name: "🐺 Волк", health: 40, damage: 15, drops: { "Еда": 3 } }, { name: "🐻 Медведь", health: 80, damage: 25, drops: { "Еда": 5, "Ткань": 3 } } ]; // Рецепты крафта const recipes = { "Каменный нож": { resources: { "Камень": 2, "Древесина": 1 }, result: { "Каменный нож": 1 } }, "Топор": { resources: { "Древесина": 3, "Камень": 1 }, result: { "Топор": 1 } }, "Лук": { resources: { "Древесина": 5, "Ткань": 2 }, result: { "Лук": 1 } }, "Еда": { resources: { "Ягоды": 5 }, result: { "Еда": 1 } } }; // Добавление сообщения в лог function addLog(message, type = "normal") { const logContainer = document.getElementById("logMessages"); const logEntry = document.createElement("div"); logEntry.className = `log-entry ${type}`; logEntry.textContent = `[${new Date().toLocaleTimeString()}] ${message}`; logContainer.appendChild(logEntry); logContainer.scrollTop = logContainer.scrollHeight; // Ограничиваем количество сообщений while (logContainer.children.length > 50) { logContainer.removeChild(logContainer.firstChild); } } // Обновление UI function updateUI() { const p = gameState.player; // Обновляем здоровье const healthPercent = (p.health / p.maxHealth) * 100; document.getElementById("healthFill").style.width = `${healthPercent}%`; document.getElementById("healthValue").textContent = `${Math.floor(p.health)}/${p.maxHealth}`; // Обновляем голод const hungerPercent = p.hunger; document.getElementById("hungerFill").style.width = `${hungerPercent}%`; document.getElementById("hungerValue").textContent = `${Math.floor(p.hunger)}/100`; // Обновляем жажду const thirstPercent = p.thirst; document.getElementById("thirstFill").style.width = `${thirstPercent}%`; document.getElementById("thirstValue").textContent = `${Math.floor(p.thirst)}/100`; // Обновляем опыт const expPercent = (p.exp / p.expToNext) * 100; document.getElementById("expFill").style.width = `${expPercent}%`; document.getElementById("expValue").textContent = `${p.exp}/${p.expToNext}`; // Обновляем уровень document.getElementById("playerLevel").textContent = p.level; // Обновляем инвентарь updateInventory(); } // Обновление инвентаря function updateInventory() { const inventoryDiv = document.getElementById("inventory"); inventoryDiv.innerHTML = ""; for (const [item, count] of Object.entries(gameState.inventory)) { if (count > 0) { const itemDiv = document.createElement("div"); itemDiv.className = "inventory-item"; itemDiv.onclick = () => useItem(item); itemDiv.innerHTML = `
${getItemIcon(item)} ${item}
${count}
`; inventoryDiv.appendChild(itemDiv); } } if (inventoryDiv.children.length === 0) { inventoryDiv.innerHTML = '
Пусто
'; } } // Получить иконку предмета function getItemIcon(item) { const icons = { "Древесина": "🪵", "Камень": "🪨", "Ягоды": "🍓", "Ткань": "🧵", "Каменный нож": "🔪", "Еда": "🍖", "Лук": "🏹", "Топор": "🪓" }; return icons[item] || "📦"; } // Использование предмета function useItem(item) { if (item === "Еда") { eat(); } else { addLog(`Вы осмотрели ${item}, но не нашли применения`, "warning"); } } // Сбор ресурсов function gather(resource) { const resources = { wood: { name: "Древесина", amount: 10, tool: "Топор", bonus: 20 }, stone: { name: "Камень", amount: 8, tool: "Каменный нож", bonus: 16 }, berries: { name: "Ягоды", amount: 6, tool: null, bonus: 0 }, cloth: { name: "Ткань", amount: 4, tool: "Каменный нож", bonus: 8 } }; const res = resources[resource]; let amount = res.amount; // Проверяем наличие инструмента const hasTool = gameState.inventory[res.tool] > 0; if (hasTool && res.tool) { amount = res.bonus; addLog(`Вы использовали ${res.tool} для сбора! +${amount} ${res.name}`, "success"); } else if (res.tool) { addLog(`Совет: используйте ${res.tool} для двойной добычи!`, "warning"); } gameState.inventory[res.name] = (gameState.inventory[res.name] || 0) + amount; addLog(`+${amount} ${res.name}`, "success"); // Уменьшаем голод и жажду при работе gameState.player.hunger = Math.max(0, gameState.player.hunger - 2); gameState.player.thirst = Math.max(0, gameState.player.thirst - 1); // Шанс на редкий предмет if (Math.random() < 0.15) { const rareItems = ["Древесный уголь", "Металлолом", "Кремень"]; const rare = rareItems[Math.floor(Math.random() * rareItems.length)]; gameState.inventory[rare] = (gameState.inventory[rare] || 0) + 1; addLog(`✨ РЕДКИЙ ДРОП! Найден ${rare}! ✨`, "success"); } checkSurvival(); updateUI(); } // Охота function hunt() { const animal = animals[Math.floor(Math.random() * animals.length)]; addLog(`⚔️ Вы встретили ${animal.name}!`, "danger"); let weapon = "руки"; let damage = 10; if (gameState.inventory["Лук"] > 0) { weapon = "лук"; damage = 35; } else if (gameState.inventory["Каменный нож"] > 0) { weapon = "нож"; damage = 25; } else if (gameState.inventory["Топор"] > 0) { weapon = "топор"; damage = 20; } addLog(`Вы атакуете ${animal.name} с ${weapon}! (урон: ${damage})`, "normal"); let animalHealth = animal.health; let playerHealth = gameState.player.health; let turn = 1; while (animalHealth > 0 && playerHealth > 0 && turn < 10) { animalHealth -= damage; addLog(`Раунд ${turn}: Вы нанесли ${damage} урона! У ${animal.name} осталось ${Math.max(0, animalHealth)} HP`, "normal"); if (animalHealth > 0) { const animalDamage = animal.damage; playerHealth -= animalDamage; addLog(`⚠️ ${animal.name} нанес ${animalDamage} урона! У вас ${Math.max(0, playerHealth)} HP`, "danger"); } turn++; } if (animalHealth <= 0) { addLog(`🎉 Вы победили ${animal.name}! 🎉`, "success"); for (const [item, count] of Object.entries(animal.drops)) { gameState.inventory[item] = (gameState.inventory[item] || 0) + count; addLog(`📦 Добыто: ${item} x${count}`, "success"); } addExp(40); gameState.player.health = playerHealth; } else { addLog(`💀 Вы проиграли бой с ${animal.name}!`, "danger"); gameState.player.health = Math.max(0, playerHealth); takeDamage(animal.damage * 2); } checkSurvival(); updateUI(); } // Показать меню крафта function showCraftMenu() { const modal = document.getElementById("craftModal"); const craftList = document.getElementById("craftList"); craftList.innerHTML = ""; for (const [item, recipe] of Object.entries(recipes)) { const craftItem = document.createElement("div"); craftItem.className = "craft-item"; let costText = ""; for (const [res, count] of Object.entries(recipe.resources)) { costText += `${res}: ${count} `; } craftItem.innerHTML = `
${getItemIcon(item)} ${item}
Требуется: ${costText}
`; craftItem.onclick = () => craft(item); craftList.appendChild(craftItem); } modal.style.display = "block"; } // Закрыть меню крафта function closeCraftModal() { document.getElementById("craftModal").style.display = "none"; } // Крафт предмета function craft(item) { const recipe = recipes[item]; if (!recipe) return; // Проверяем ресурсы let canCraft = true; for (const [res, count] of Object.entries(recipe.resources)) { if ((gameState.inventory[res] || 0) < count) { canCraft = false; addLog(`❌ Недостаточно ${res} для создания ${item}!`, "warning"); break; } } if (canCraft) { // Забираем ресурсы for (const [res, count] of Object.entries(recipe.resources)) { gameState.inventory[res] -= count; if (gameState.inventory[res] === 0) { delete gameState.inventory[res]; } } // Добавляем предмет for (const [newItem, count] of Object.entries(recipe.result)) { gameState.inventory[newItem] = (gameState.inventory[newItem] || 0) + count; } addLog(`✅ Создан ${item}!`, "success"); addExp(15); updateUI(); } closeCraftModal(); } // Поесть function eat() { if (gameState.inventory["Еда"] > 0) { gameState.inventory["Еда"]--; gameState.player.hunger = Math.min(100, gameState.player.hunger + 30); gameState.player.health = Math.min(gameState.player.maxHealth, gameState.player.health + 15); addLog(`🍖 Вы съели еду. Голод: ${Math.floor(gameState.player.hunger)}/100`, "success"); updateUI(); } else if (gameState.inventory["Ягоды"] > 0) { gameState.inventory["Ягоды"]--; gameState.player.hunger = Math.min(100, gameState.player.hunger + 10); addLog(`🍓 Вы съели ягоды. Голод: ${Math.floor(gameState.player.hunger)}/100`, "success"); updateUI(); } else { addLog(`❌ Нет еды в инвентаре!`, "warning"); } } // Выпить воду function drink() { gameState.player.thirst = Math.min(100, gameState.player.thirst + 30); addLog(`💧 Вы напились воды. Жажда: ${Math.floor(gameState.player.thirst)}/100`, "success"); updateUI(); } // Исследовать function explore() { addLog(`🚶 Вы отправляетесь исследовать окрестности...`, "normal"); const events = [ { text: "Вы нашли заброшенную базу с припасами! +20 опыта", type: "success", exp: 20, health: 0 }, { text: "На вас напал другой выживший! -15 здоровья", type: "danger", exp: 0, health: -15 }, { text: "Вы нашли ящик с ресурсами! +10 древесины", type: "success", exp: 0, wood: 10 }, { text: "Вас укусила змея! -10 здоровья", type: "danger", exp: 0, health: -10 }, { text: "Вы нашли лекарственные травы! +15 здоровья", type: "success", exp: 0, health: 15 } ]; const event = events[Math.floor(Math.random() * events.length)]; addLog(event.text, event.type); if (event.exp) addExp(event.exp); if (event.health) { gameState.player.health = Math.min(gameState.player.maxHealth, gameState.player.health + event.health); if (event.health < 0) takeDamage(-event.health); } if (event.wood) { gameState.inventory["Древесина"] = (gameState.inventory["Древесина"] || 0) + event.wood; } gameState.player.hunger = Math.max(0, gameState.player.hunger - 3); gameState.player.thirst = Math.max(0, gameState.player.thirst - 4); checkSurvival(); updateUI(); } // Отдохнуть function rest() { addLog(`🛌 Вы отдыхаете у костра...`, "normal"); gameState.player.health = Math.min(gameState.player.maxHealth, gameState.player.health + 20); gameState.player.hunger = Math.max(0, gameState.player.hunger - 5); gameState.player.thirst = Math.max(0, gameState.player.thirst - 5); addLog(`❤️ Вы восстановили 20 здоровья`, "success"); checkSurvival(); updateUI(); } // Получить опыт function addExp(amount) { gameState.player.exp += amount; addLog(`⭐ +${amount} опыта!`, "success"); while (gameState.player.exp >= gameState.player.expToNext) { gameState.player.exp -= gameState.player.expToNext; gameState.player.level++; gameState.player.maxHealth += 10; gameState.player.health = gameState.player.maxHealth; gameState.player.expToNext = gameState.player.level * 100; addLog(`🎉 УРОВЕНЬ ПОВЫШЕН! Теперь вы ${gameState.player.level} уровень! 🎉`, "success"); addLog(`❤️ Максимальное здоровье увеличено до ${gameState.player.maxHealth}!`, "success"); } } // Получить урон function takeDamage(amount) { gameState.player.health -= amount; if (gameState.player.health <= 0) { gameState.player.health = 50; gameState.player.hunger = 80; gameState.player.thirst = 80; addLog(`💀 ВЫ УМЕРЛИ! Возрождение с потерей половины ресурсов... 💀`, "danger"); // Потеря половины ресурсов for (const [item, count] of Object.entries(gameState.inventory)) { if (count > 1) { gameState.inventory[item] = Math.floor(count / 2); } else { delete gameState.inventory[item]; } } } } // Проверить выживание (голод/жажда) function checkSurvival() { if (gameState.player.hunger <= 0) { addLog(`⚠️ ВЫ УМИРАЕТЕ ОТ ГОЛОДА!`, "danger"); takeDamage(5); } if (gameState.player.thirst <= 0) { addLog(`⚠️ ВЫ УМИРАЕТЕ ОТ ЖАЖДЫ!`, "danger"); takeDamage(5); } } // Тики времени (каждые 30 секунд) setInterval(() => { gameState.player.hunger = Math.max(0, gameState.player.hunger - 2); gameState.player.thirst = Math.max(0, gameState.player.thirst - 3); checkSurvival(); updateUI(); }, 30000); // Инициализация игры function init() { updateUI(); addLog("Игра загружена! Удачной охоты!", "success"); } // Запуск init();
